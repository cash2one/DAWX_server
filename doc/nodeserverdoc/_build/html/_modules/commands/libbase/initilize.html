[docs]def init_syspath():
    """
    增加 g.path 到 /etc/profile
    """
    PATH=r"/etc/profile"
    if not g.name and not g.path:
        print "not found g.name and g.path ,please set it "
        sys.exit(100)
    astr = "export %s=%s\n"%(g.name,g.path)
    f = open(PATH,'r')
    op = f.readlines()
    tag = True
    for line in op:
        if g.name in line :
            op.remove(line)
            tag = False
    f.close()

    if  tag:
        f = open(PATH,'w')
        op.append(astr)
        f.writelines(op)
        f.close()
